- name: Install and Start NGINX on Web Servers
  hosts: host1
  become: true

  tasks:
    - name: Update APT cache
      apt:
        update_cache: yes

    - name: Install NGINX
      apt:
        name: nginx
        state: present

    - name: Ensure NGINX is running
      service:
        name: nginx
        state: started
        enabled: yes
